import Tabs from '@theme/Tabs';

export const DownloadButton = ({href, children, primary}) => (
  <a
    href={href}
    className={`download-button ${primary ? 'download-button--primary' : ''}`}
    target="_blank"
    rel="noopener noreferrer">
    {children}
  </a>
);

export const DownloadButtonGroup = ({children}) => (
  <div className="download-button-group">
    {children}
  </div>
);

export const OSDetector = () => {
  useEffect(() => {
    const detectOS = () => {
      const ua = navigator.userAgent.toLowerCase();
      const platform = navigator.platform?.toLowerCase() || '';
      if (ua.includes('win')) return 'windows';
      if (ua.includes('mac') || platform.includes('mac')) return 'macos';
      if (ua.includes('linux') || platform.includes('linux')) return 'linux';
      return null;
    };

    const os = detectOS();
    if (!os) return;

    const tabMap = {
      'macos': 'macOS',
      'linux': 'Linux',
      'windows': 'Windows'
    };
    const tabLabel = tabMap[os];
    if (!tabLabel) return;

    setTimeout(() => {
      const tabs = document.querySelectorAll('[role="tab"]');
      tabs.forEach(tab => {
        if (tab.textContent === tabLabel) {
          tab.click();
        }
      });
    }, 100);

}, []);

return null;
};

### Installation

```bash
# Download for your architecture (example: Apple Silicon)
curl -L -o muti-metroo https://download.mutimetroo.com/darwin-arm64/muti-metroo

# Make executable and install
chmod +x muti-metroo
sudo mv muti-metroo /usr/local/bin/

# Verify installation
muti-metroo --version
```

:::note Gatekeeper Warning
On first run, macOS may show a security warning because the binary is not signed with a Developer ID certificate. To allow it:
1. Right-click the binary and select "Open", or
2. Go to System Preferences > Security & Privacy and click "Allow Anyway"
:::

## Verify Your Download

Verify your download with SHA256 checksums:

<DownloadButtonGroup>
  <DownloadButton href="https://download.mutimetroo.com/muti-metroo/checksums.txt">Download Checksums</DownloadButton>
</DownloadButtonGroup>

<Tabs groupId="verify-os">
  <TabItem value="unix" label="Linux / macOS" default>

```bash
# Download checksums
curl -L -o checksums.txt https://download.mutimetroo.com/muti-metroo/checksums.txt

# Verify (example for linux-amd64)
sha256sum -c checksums.txt --ignore-missing
```

  </TabItem>
  <TabItem value="windows" label="Windows">

```powershell
# Get the expected hash from checksums.txt
$expected = (Get-Content checksums.txt | Select-String "muti-metroo-windows-amd64.exe").ToString().Split()[0]

# Calculate actual hash
$actual = (Get-FileHash muti-metroo-windows-amd64.exe -Algorithm SHA256).Hash.ToLower()

# Compare
if ($expected -eq $actual) { "OK" } else { "MISMATCH" }
```
